class Solution {
public:
    vector<vector<string>> groupAnagrams(vector<string>& strs) {
        vector<vector<string>> re;//结果
        unordered_multimap<unordered_map<char,short>,string>strsTable;//总表
        for (string str : strs) {   //遍历数组
            unordered_map<char, short> strTable; // 字母表
            for (char c : str) {  //遍历单个str中的char
                ++strTable[c];
            }
            if (strsTable.find(strTable)!=strsTable.end()) { // 判断是否在总表，不在则单开
                
            }
            //插入
        }
        //将结果存储到re中
        
        return re;
    }
};
```

```
思路：每个单词对应一个单词表，单词表记录每个单词中每个字母的个数。异位词的单词表相等，记为一组。否则则重新另开一组。故需要一份总表，记录每个单词的单词表以供查询。
